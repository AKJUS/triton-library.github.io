84 std::string SymbolicExpression::getFormattedId(void) const {
85 if (this->ast == nullptr)
86 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedId(): No AST defined.");
87
88 if (ast->getContext()->getRepresentationMode() == triton::ast::representations::SMT_REPRESENTATION)
89 return "ref!" + std::to_string(this->id);
90
91 else if (ast->getContext()->getRepresentationMode() == triton::ast::representations::PYTHON_REPRESENTATION)
92 return "ref_" + std::to_string(this->id);
93
94 else if (ast->getContext()->getRepresentationMode() == triton::ast::representations::PCODE_REPRESENTATION) {
95 if (this->isMemory()) {
96 auto mem = this->getOriginMemory();
97 std::ostringstream ss;
98 ss << "@[0x" << std::hex << mem.getAddress() << ":" << std::dec << mem.getBitSize() << "]";
99 return ss.str();
100 }
101 else if (this->isRegister()) {
102 auto reg = this->getOriginRegister();
103 return reg.getName() + "_" + std::to_string(this->id);
104 }
105 else {
106 return "tmp_" + std::to_string(this->id);
107 }
108 }
109
110 else
111 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedId(): Invalid AST representation mode.");
112 }

115 std::string SymbolicExpression::getFormattedComment(void) const {
116 if (this->ast == nullptr)
117 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedComment(): No AST defined.");
118
119 if (this->getComment().empty())
120 return "";
121
122 switch (ast->getContext()->getRepresentationMode()) {
123 case triton::ast::representations::SMT_REPRESENTATION:
124 case triton::ast::representations::PCODE_REPRESENTATION:
125 return "; " + this->getComment();
126
127 case triton::ast::representations::PYTHON_REPRESENTATION:
128 return "# " + this->getComment();
129
130 default:
131 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedComment(): Invalid AST representation mode.");
132 };
133 }

157 std::string SymbolicExpression::getFormattedExpression(void) const {
158 std::ostringstream stream;
159
160 if (this->ast == nullptr)
161 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedExpression(): No AST defined.");
162
163 switch (ast->getContext()->getRepresentationMode()) {
164 case triton::ast::representations::SMT_REPRESENTATION:
165 stream << (this->getAst()->isArray() ? this->getArrayDefine() : this->getBitvectorDefine());
166 break;
167 case triton::ast::representations::PCODE_REPRESENTATION:
168 case triton::ast::representations::PYTHON_REPRESENTATION:
169 stream << this->getFormattedId() << " = " << this->getAst();
170 break;
171
172 default:
173 throw triton::exceptions::SymbolicExpression("SymbolicExpression::getFormattedExpression(): Invalid AST representation mode.");
174 }
175
176 if (!this->getComment().empty()) {
177 stream << " " << this->getFormattedComment();
178 }
179
180 return stream.str();
181 }

199 void SymbolicExpression::setAst(const triton::ast::SharedAbstractNode& node) {
200 triton::ast::SharedAbstractNode old = this->ast;
201
202 /* If node is the same as the old one, just do not set the ast. */
203 if (node == old)
204 return;
205
206 if (old) {
207 /* Link old parents with the new node */
208 for (auto sp : old->getParents()) {
209 node->setParent(sp.get());
210 }
211 }
212
213 /* Set the new ast */
214 this->ast = node;
215
216 /* Do not init parents if the new node has same properties that the old one */
217 if (!old || !old->canReplaceNodeWithoutUpdate(ast)) {
218 this->ast->initParents();
219 }
220 }
