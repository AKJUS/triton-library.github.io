34  void TritonToLLVM::createFunction(const triton::ast::SharedAbstractNode& node, const std::string& fname) {
35  // Collect used symbolic variables.
36  auto vars = triton::ast::search(node, triton::ast::VARIABLE_NODE);
37 
38  // Each symbolic variable is a function argument
39  std::vector<llvm::Type*> argsType;
40  argsType.resize(vars.size());
41  for (triton::usize index = 0 ; index < vars.size() ; index++) {
42  switch (vars[index]->getBitvectorSize()) {
43  case 8:
44  argsType[index] = llvm::Type::getInt8Ty(this->llvmContext);
45  break;
46  case 16:
47  argsType[index] = llvm::Type::getInt16Ty(this->llvmContext);
48  break;
49  case 32:
50  argsType[index] = llvm::Type::getInt32Ty(this->llvmContext);
51  break;
52  case 64:
53  argsType[index] = llvm::Type::getInt64Ty(this->llvmContext);
54  break;
55  default:
56  throw triton::exceptions::AstLifting("TritonToLLVM::do_convert(): Symbolic variables must be aligned on 8, 16, 32 or 64 bit.");
57  }
58  }
59 
60  /* Declare LLVM function */
61  auto retSize = node->getBitvectorSize();
62  auto* retType = llvm::IntegerType::get(this->llvmContext, retSize);
63  auto* funcType = llvm::FunctionType::get(retType, argsType, false /* isVarArg */);
64  auto* llvmFunc = llvm::Function::Create(funcType, llvm::Function::ExternalLinkage, fname, this->llvmModule.get());
65 
66  /* Rename parameters */
67  llvm::Function::arg_iterator params = llvmFunc->arg_begin();
68  for (const auto& node : vars) {
69  auto var = reinterpret_cast<triton::ast::VariableNode*>(node.get())->getSymbolicVariable();
70  auto* param = params++;
71  param->setName(var->getName());
72  this->llvmVars[node] = param;
73  }
74 
75  // A Triton expression is represented as one basic block
76  auto* llvmBasicBlock = llvm::BasicBlock::Create(this->llvmContext, "entry", llvmFunc);
77  this->llvmIR.SetInsertPoint(llvmBasicBlock);
78  }

81  std::shared_ptr<llvm::Module> TritonToLLVM::convert(const triton::ast::SharedAbstractNode& node, const std::string& fname) {
82  std::unordered_map<triton::ast::SharedAbstractNode, llvm::Value*> results;
83 
84  /* Create the LLVM function */
85  this->createFunction(node, fname);
86 
87  /* Lift Triton AST to LLVM IR */
88  auto nodes = triton::ast::childrenExtraction(node, true /* unroll*/, true /* revert */);
89  for (const auto& node : nodes) {
90  if (node->getBitvectorSize()) {
91  results.insert(std::make_pair(node, this->do_convert(node, &results)));
92  }
93  }
94 
95  /* Create the return instruction */
96  this->llvmIR.CreateRet(results.at(node));
97 
98  return this->llvmModule;
99  }

102  llvm::Value* TritonToLLVM::do_convert(const triton::ast::SharedAbstractNode& node, std::unordered_map<triton::ast::SharedAbstractNode, llvm::Value*>* results) {
103  if (node == nullptr)
104  throw triton::exceptions::AstLifting("TritonToLLVM::do_convert(): node cannot be null.");
105 
106  /* Prepare llvm's children */
107  std::vector<llvm::Value*> children;
108  for (auto&& n : node->getChildren()) {
109  /* Ignore children like INTEGER_NODE */
110  if (n->getBitvectorSize() == 0) {
111  children.emplace_back(nullptr);
112  }
113  else {
114  children.emplace_back(results->at(n));
115  }
116  }
117 
118  switch (node->getType()) {
119  case triton::ast::BVADD_NODE:
120  return this->llvmIR.CreateAdd(children[0], children[1]);
121 
122  case triton::ast::BVAND_NODE:
123  return this->llvmIR.CreateAnd(children[0], children[1]);
124 
125  case triton::ast::BVASHR_NODE:
126  return this->llvmIR.CreateAShr(children[0], children[1]);
127 
128  case triton::ast::BVLSHR_NODE:
129  return this->llvmIR.CreateLShr(children[0], children[1]);
130 
131  case triton::ast::BVMUL_NODE:
132  return this->llvmIR.CreateMul(children[0], children[1]);
133 
134  case triton::ast::BVNAND_NODE:
135  return this->llvmIR.CreateNot(this->llvmIR.CreateAnd(children[0], children[1]));
136 
137  case triton::ast::BVNEG_NODE:
138  return this->llvmIR.CreateNeg(children[0]);
139 
140  case triton::ast::BVNOR_NODE:
141  return this->llvmIR.CreateNot(this->llvmIR.CreateOr(children[0], children[1]));
142 
143  case triton::ast::BVNOT_NODE:
144  return this->llvmIR.CreateNot(children[0]);
145 
146  case triton::ast::BVOR_NODE:
147  return this->llvmIR.CreateOr(children[0], children[1]);
148 
149  // bvrol(expr, rot) = ((expr << (rot % size)) | (expr >> (size - (rot % size))))
150  case triton::ast::BVROL_NODE: {
151  auto rot = reinterpret_cast<triton::ast::IntegerNode*>(node->getChildren()[1].get())->getInteger().convert_to<uint64_t>();
152  auto size = node->getBitvectorSize();
153  return this->llvmIR.CreateOr(this->llvmIR.CreateShl(children[0], rot % size), this->llvmIR.CreateLShr(children[0], (size - (rot % size))));
154  }
155 
156  // bvror(expr, rot) = ((expr >> (rot % size)) | (expr << (size - (rot % size))))
157  case triton::ast::BVROR_NODE: {
158  auto rot = reinterpret_cast<triton::ast::IntegerNode*>(node->getChildren()[1].get())->getInteger().convert_to<uint64_t>();
159  auto size = node->getBitvectorSize();
160  return this->llvmIR.CreateOr(this->llvmIR.CreateLShr(children[0], rot % size), this->llvmIR.CreateShl(children[0], (size - (rot % size))));
161  }
162 
163  case triton::ast::BVSDIV_NODE:
164  return this->llvmIR.CreateSDiv(children[0], children[1]);
165 
166  case triton::ast::BVSGE_NODE:
167  return this->llvmIR.CreateICmpSGE(children[0], children[1]);
168 
169  case triton::ast::BVSGT_NODE:
170  return this->llvmIR.CreateICmpSGT(children[0], children[1]);
171 
172  case triton::ast::BVSHL_NODE:
173  return this->llvmIR.CreateShl(children[0], children[1]);
174 
175  case triton::ast::BVSLE_NODE:
176  return this->llvmIR.CreateICmpSLE(children[0], children[1]);
177 
178  case triton::ast::BVSLT_NODE:
179  return this->llvmIR.CreateICmpSLT(children[0], children[1]);
180 
181  case triton::ast::BVSMOD_NODE: {
182  auto* LHS = children[0];
183  auto* RHS = children[1];
184  return this->llvmIR.CreateSRem(this->llvmIR.CreateAdd(this->llvmIR.CreateSRem(LHS, RHS), RHS), RHS);
185  }
186 
187  case triton::ast::BVSREM_NODE:
188  return this->llvmIR.CreateSRem(children[0], children[1]);
189 
190  case triton::ast::BVSUB_NODE:
191  return this->llvmIR.CreateSub(children[0], children[1]);
192 
193  case triton::ast::BVUDIV_NODE:
194  return this->llvmIR.CreateUDiv(children[0], children[1]);
195 
196  case triton::ast::BVUGE_NODE:
197  return this->llvmIR.CreateICmpUGE(children[0], children[1]);
198 
199  case triton::ast::BVUGT_NODE:
200  return this->llvmIR.CreateICmpUGT(children[0], children[1]);
201 
202  case triton::ast::BVULE_NODE:
203  return this->llvmIR.CreateICmpULE(children[0], children[1]);
204 
205  case triton::ast::BVULT_NODE:
206  return this->llvmIR.CreateICmpULT(children[0], children[1]);
207 
208  case triton::ast::BVUREM_NODE:
209  return this->llvmIR.CreateURem(children[0], children[1]);
210 
211  case triton::ast::BVXNOR_NODE:
212  return this->llvmIR.CreateNot(this->llvmIR.CreateXor(children[0], children[1]));
213 
214  case triton::ast::BVXOR_NODE:
215  return this->llvmIR.CreateXor(children[0], children[1]);
216 
217  case triton::ast::BV_NODE:
218  return llvm::ConstantInt::get(this->llvmContext, llvm::APInt(node->getBitvectorSize(), node->evaluate().convert_to<uint64>(), false));
219 
220  case triton::ast::CONCAT_NODE: {
221  auto dstSize = node->getBitvectorSize();
222  auto finalNode = this->llvmIR.CreateZExt(children[0], llvm::IntegerType::get(this->llvmContext, dstSize));
223 
224  for (triton::usize index = 1; index < children.size(); index++) {
225  finalNode = this->llvmIR.CreateShl(finalNode, node->getChildren()[index]->getBitvectorSize());
226  auto* n = this->llvmIR.CreateZExt(children[index], llvm::IntegerType::get(this->llvmContext, dstSize));
227  finalNode = this->llvmIR.CreateOr(finalNode, n);
228  }
229 
230  return finalNode;
231  }
232 
233  case triton::ast::DISTINCT_NODE:
234  return this->llvmIR.CreateICmpNE(children[0], children[1]);
235 
236  case triton::ast::EQUAL_NODE:
237  return this->llvmIR.CreateICmpEQ(children[0], children[1]);
238 
239  case triton::ast::EXTRACT_NODE: {
240  auto low = reinterpret_cast<triton::ast::IntegerNode*>(node->getChildren()[1].get())->getInteger().convert_to<uint64_t>();
241  auto dstSize = node->getChildren()[2]->getBitvectorSize();
242  auto* value = children[2];
243 
244  if (low == 0) {
245  return this->llvmIR.CreateTrunc(value, llvm::IntegerType::get(this->llvmContext, node->getBitvectorSize()));
246  }
247 
248  return this->llvmIR.CreateTrunc(this->llvmIR.CreateLShr(value, low), llvm::IntegerType::get(this->llvmContext, node->getBitvectorSize()));
249  }
250 
251  case triton::ast::ITE_NODE:
252  return this->llvmIR.CreateSelect(children[0], children[1], children[2]);
253 
254  case triton::ast::LAND_NODE: {
255  auto* truenode = llvm::ConstantInt::get(this->llvmContext, llvm::APInt(1, 1));
256  return this->llvmIR.CreateICmpEQ(this->llvmIR.CreateAnd(children), truenode);
257  }
258 
259  case triton::ast::LNOT_NODE: {
260  auto* truenode = llvm::ConstantInt::get(this->llvmContext, llvm::APInt(1, 1));
261  return this->llvmIR.CreateICmpEQ(this->llvmIR.CreateNot(children[0]), truenode);
262  }
263 
264  case triton::ast::LOR_NODE: {
265  auto* truenode = llvm::ConstantInt::get(this->llvmContext, llvm::APInt(1, 1));
266  return this->llvmIR.CreateICmpEQ(this->llvmIR.CreateOr(children), truenode);
267  }
268 
269  case triton::ast::LXOR_NODE: {
270  auto* child0 = children[0];
271  auto* child1 = children[1];
272  auto* current = this->llvmIR.CreateXor(child0, child1);
273 
274  for (triton::usize index = 2; index < children.size(); index++) {
275  current = this->llvmIR.CreateXor(current, children[index]);
276  }
277 
278  auto* truenode = llvm::ConstantInt::get(this->llvmContext, llvm::APInt(1, 1));
279  return this->llvmIR.CreateICmpEQ(current, truenode);
280  }
281 
282  case triton::ast::REFERENCE_NODE:
283  return results->at(reinterpret_cast<triton::ast::ReferenceNode*>(node.get())->getSymbolicExpression()->getAst());
284 
285  case triton::ast::SX_NODE:
286  return this->llvmIR.CreateSExt(children[1], llvm::IntegerType::get(this->llvmContext, node->getBitvectorSize()));
287 
288  case triton::ast::VARIABLE_NODE:
289  return this->llvmVars.at(node);
290 
291  case triton::ast::ZX_NODE:
292  return this->llvmIR.CreateZExt(children[1], llvm::IntegerType::get(this->llvmContext, node->getBitvectorSize()));
293 
294  default:
295  throw triton::exceptions::AstLifting("TritonToLLVM::do_convert(): Invalid kind of node.");
296  }
297  }
