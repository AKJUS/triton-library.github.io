213 template <typename T> SharedAbstractNode concat(const T& exprs) {
214 SharedAbstractNode node = std::make_shared<ConcatNode>(exprs, this->shared_from_this());
215 if (node == nullptr)
216 throw triton::exceptions::Ast("Node builders - Not enough memory");
217 node->init();
218
219 if (this->modes->isModeEnabled(triton::modes::CONSTANT_FOLDING)) {
220 if (node->isSymbolized() == false) {
221 return this->bv(node->evaluate(), node->getBitvectorSize());
222 }
223 }
224
225 if (this->modes->isModeEnabled(triton::modes::AST_OPTIMIZATIONS)) {
226 /* Optimization: concatenate extractions in one if possible */
227 auto n = this->simplify_concat(std::vector<SharedAbstractNode>(exprs.begin(), exprs.end()));
228 if (n) {
229 return n;
230 }
231 }
232
233 return this->collect(node);
234 }
