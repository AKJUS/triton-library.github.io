180 triton::ast::SharedAbstractNode SymbolicSimplification::simplify(const triton::ast::SharedAbstractNode& node) const {
181 std::list<triton::ast::SharedAbstractNode> worklist;
182 triton::ast::SharedAbstractNode snode = node;
183
184 if (node == nullptr)
185 throw triton::exceptions::SymbolicSimplification("SymbolicSimplification::simplify(): node cannot be null.");
186
187 if (this->callbacks && this->callbacks->isDefined(triton::callbacks::SYMBOLIC_SIMPLIFICATION)) {
188 snode = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, node);
189 /*
190 * We use a worklist strategy to avoid recursive calls
191 * and so stack overflow when going through a big AST.
192 */
193 worklist.push_back(snode);
194 while (worklist.size()) {
195 auto ast = worklist.front();
196 worklist.pop_front();
197 bool needs_update = false;
198 for (triton::uint32 index = 0; index < ast->getChildren().size(); index++) {
199 auto child = ast->getChildren()[index];
200 /* Don't apply simplification on nodes like String, Integer, etc. */
201 if (child->getBitvectorSize()) {
202 auto schild = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, child);
203 ast->setChild(index, schild);
204 needs_update |= !schild->canReplaceNodeWithoutUpdate(child);
205 worklist.push_back(schild);
206 }
207 }
208 if (needs_update) {
209 ast->init(true);
210 }
211 }
212 }
213
214 return snode;
215 }

223 triton::arch::BasicBlock SymbolicSimplification::deadStoreElimination(const triton::arch::BasicBlock& block, bool padding) const {
224 std::unordered_map<triton::usize, SharedSymbolicExpression> lifetime;
225 std::map<triton::uint64, triton::arch::Instruction> instructions;
226 triton::arch::BasicBlock out;
227 triton::arch::BasicBlock in = block;
228
229 if (block.getSize() == 0)
230 return {};
231
232 /* Define a temporary Context */
233 triton::Context tmpctx(in.getInstructions()[0].getArchitecture());
234
235 /* Execute the block */
236 tmpctx.processing(in);
237
238 /* Get all symbolic registers that were written */
239 for (auto& reg : tmpctx.getSymbolicRegisters()) {
240 for (auto& item : tmpctx.sliceExpressions(reg.second)) {
241 lifetime[item.first] = item.second;
242 }
243 }
244
245 /* Get all symbolic memory cells that were written */
246 for (auto& mem : tmpctx.getSymbolicMemory()) {
247 for (auto& item : tmpctx.sliceExpressions(mem.second)) {
248 lifetime[item.first] = item.second;
249 }
250 }
251
252 /* Get back the origin assembly of expressions that still alive */
253 for (auto& se : lifetime) {
254 auto assembly = se.second->getDisassembly();
255 if (assembly.empty())
256 continue;
257 triton::uint64 addr = std::stoi(assembly, nullptr, 16);
258 for (auto& inst : in.getInstructions()) {
259 if (inst.getAddress() == addr) {
260 instructions[addr] = inst;
261 break;
262 }
263 }
264 }
265
266 /* Create a new block with sorted instructions */
267 auto lastaddr = in.getFirstAddress();
268 auto nop = tmpctx.getNopInstruction();
269 for (auto& item : instructions) {
270 if (padding) {
271 while (item.second.getAddress() > lastaddr) {
272 out.add(nop);
273 lastaddr += nop.getSize();
274 }
275 }
276 out.add(item.second);
277 lastaddr = item.second.getNextAddress();
278 }
279
280 return out;
281 }
