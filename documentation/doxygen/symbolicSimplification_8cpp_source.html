179  triton::ast::SharedAbstractNode SymbolicSimplification::simplify(const triton::ast::SharedAbstractNode& node) const {
180  std::list<triton::ast::SharedAbstractNode> worklist;
181  triton::ast::SharedAbstractNode snode = node;
182 
183  if (node == nullptr)
184  throw triton::exceptions::SymbolicSimplification("SymbolicSimplification::simplify(): node cannot be null.");
185 
186  if (this->callbacks && this->callbacks->isDefined(triton::callbacks::SYMBOLIC_SIMPLIFICATION)) {
187  snode = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, node);
188  /*
189  * We use a worklist strategy to avoid recursive calls
190  * and so stack overflow when going through a big AST.
191  */
192  worklist.push_back(snode);
193  while (worklist.size()) {
194  auto ast = worklist.front();
195  worklist.pop_front();
196  bool needs_update = false;
197  for (triton::uint32 index = 0; index < ast->getChildren().size(); index++) {
198  auto child = ast->getChildren()[index];
199  /* Don't apply simplification on nodes like String, Integer, etc. */
200  if (child->getBitvectorSize()) {
201  auto schild = this->callbacks->processCallbacks(triton::callbacks::SYMBOLIC_SIMPLIFICATION, child);
202  ast->setChild(index, schild);
203  needs_update |= !schild->canReplaceNodeWithoutUpdate(child);
204  worklist.push_back(schild);
205  }
206  }
207  if (needs_update) {
208  ast->init(true);
209  }
210  }
211  }
212 
213  return snode;
214  }

222  triton::arch::BasicBlock SymbolicSimplification::deadStoreElimination(const triton::arch::BasicBlock& block) const {
223  std::unordered_map<triton::usize, SharedSymbolicExpression> lifetime;
224  std::map<triton::uint64, triton::arch::Instruction> instructions;
225  triton::arch::BasicBlock out;
226  triton::arch::BasicBlock in = block;
227 
228  if (block.getSize() == 0)
229  return {};
230 
231  /* Define a temporary API */
232  triton::API tmpapi(in.getInstructions()[0].getArchitecture());
233 
234  /* Execute the block */
235  tmpapi.processing(in);
236 
237  /* Get all symbolic registers that were written */
238  for (auto& reg : tmpapi.getSymbolicRegisters()) {
239  for (auto& item : tmpapi.sliceExpressions(reg.second)) {
240  lifetime[item.first] = item.second;
241  }
242  }
243 
244  /* Get all symbolic memory cells that were written */
245  for (auto& mem : tmpapi.getSymbolicMemory()) {
246  for (auto& item : tmpapi.sliceExpressions(mem.second)) {
247  lifetime[item.first] = item.second;
248  }
249  }
250 
251  /* Get back the origin assembly of expressions that still alive */
252  for (auto& se : lifetime) {
253  auto assembly = se.second->getDisassembly();
254  if (assembly.empty())
255  continue;
256  triton::uint64 addr = std::stoi(assembly, nullptr, 16);
257  for (auto& inst : in.getInstructions()) {
258  if (inst.getAddress() == addr) {
259  instructions[addr] = inst;
260  break;
261  }
262  }
263  }
264 
265  /* Create a new block with sorted instructions */
266  for (auto& item : instructions) {
267  out.add(item.second);
268  }
269 
270  return out;
271  }
