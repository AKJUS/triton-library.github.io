47 void Architecture::setArchitecture(triton::arch::architecture_e arch) {
48 /* Allocate and init the good arch */
49 switch (arch) {
50 case triton::arch::ARCH_X86_64: this->cpu.reset(new(std::nothrow) triton::arch::x86::x8664Cpu(this->callbacks)); break;
51 case triton::arch::ARCH_X86: this->cpu.reset(new(std::nothrow) triton::arch::x86::x86Cpu(this->callbacks)); break;
52 case triton::arch::ARCH_AARCH64: this->cpu.reset(new(std::nothrow) triton::arch::arm::aarch64::AArch64Cpu(this->callbacks)); break;
53 case triton::arch::ARCH_ARM32: this->cpu.reset(new(std::nothrow) triton::arch::arm::arm32::Arm32Cpu(this->callbacks)); break;
54 default:
55 throw triton::exceptions::Architecture("Architecture::setArchitecture(): Architecture not supported.");
56 }
57
58 if (this->cpu == nullptr) {
59 throw triton::exceptions::Architecture("Architecture::setArchitecture(): Not enough memory.");
60 }
61
62 /* Setup global variables */
63 this->arch = arch;
64 }

229 void Architecture::disassembly(triton::arch::BasicBlock& block, triton::uint64 addr) const {
230 if (!this->cpu)
231 throw triton::exceptions::Architecture("Architecture::disassembly(): You must define an architecture.");
232
233 for (auto& inst : block.getInstructions()) {
234 inst.setAddress(addr);
235 this->cpu->disassembly(inst);
236 addr += inst.getSize();
237 }

241 std::vector<triton::arch::Instruction> Architecture::disassembly(triton::uint64 addr, triton::usize count) const {
242 std::vector<triton::arch::Instruction> ret;
243 ret.reserve(count);
244
245 while (count--) {
246 if (!this->isConcreteMemoryValueDefined(addr)) {
247 break;
248 }
249 auto opcodes = this->getConcreteMemoryAreaValue(addr, 16);
250 auto inst = triton::arch::Instruction(addr, reinterpret_cast<triton::uint8*>(opcodes.data()), opcodes.size());
251 this->disassembly(inst);
252 ret.push_back(inst);
253 addr += inst.getSize();
254 }
255
256 return ret;
257 }

260 triton::arch::BasicBlock Architecture::disassembly(triton::uint64 addr) const {
261 std::vector<triton::arch::Instruction> ret;
262
263 do {
264 if (!this->isConcreteMemoryValueDefined(addr)) {
265 break;
266 }
267 auto opcodes = this->getConcreteMemoryAreaValue(addr, 16);
268 auto inst = triton::arch::Instruction(addr, reinterpret_cast<triton::uint8*>(opcodes.data()), opcodes.size());
269 this->disassembly(inst);
270 ret.push_back(inst);
271 addr += inst.getSize();
272 } while (!ret.back().isControlFlow());
273
274 return triton::arch::BasicBlock(ret);
275 }
