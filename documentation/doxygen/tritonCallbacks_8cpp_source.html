78 static PyObject* triton_BasicBlock(PyObject* self, PyObject* args) {
79 PyObject* obj = nullptr;
80
81 /* Extract arguments */
82 if (PyArg_ParseTuple(args, "|O", &obj) == false) {
83 return PyErr_Format(PyExc_TypeError, "BasicBlock(): Invalid constructor.");
84 }
85
86 try {
87 /* Check if it's a default constructor */
88 if (obj == nullptr)
89 return PyBasicBlock();
90
91 /* Check if argument is a list of instruction */
92 else if (obj != nullptr && PyList_Check(obj)) {
93 triton::arch::BasicBlock block;
94
95 for (Py_ssize_t i = 0; i < PyList_Size(obj); i++) {
96 PyObject* item = PyList_GetItem(obj, i);
97
98 if (PyInstruction_Check(item) == false)
99 return PyErr_Format(PyExc_TypeError, "BasicBlock(): All items must be an Instruction objet.");
100
101 block.add(*PyInstruction_AsInstruction(item));
102 }
103
104 return PyBasicBlock(block);
105 }
106
107 /* Otherwise, invalid constructor */
108 else {
109 return PyErr_Format(PyExc_TypeError, "BasicBlock(): Invalid constructor.");
110 }
111 }
112 catch (const triton::exceptions::Exception& e) {
113 return PyErr_Format(PyExc_TypeError, "%s", e.what());
114 }
115 }

144 static PyObject* triton_Instruction(PyObject* self, PyObject* args) {
145 PyObject* arg1 = nullptr;
146 PyObject* arg2 = nullptr;
147
148 /* Extract arguments */
149 if (PyArg_ParseTuple(args, "|OO", &arg1, &arg2) == false) {
150 return PyErr_Format(PyExc_TypeError, "Instruction(): Invalid constructor.");
151 }
152
153 try {
154 /* Instruction() */
155 if (arg1 == nullptr)
156 return PyInstruction();
157
158 /* Instruction(opcode) */
159 else if (arg1 && PyBytes_Check(arg1) && arg2 == nullptr) {
160 triton::uint8* opc = reinterpret_cast<triton::uint8*>(PyBytes_AsString(arg1));
161 triton::uint32 size = static_cast<triton::uint32>(PyBytes_Size(arg1));
162 return PyInstruction(opc, size);
163 }
164
165 /* Instruction(address, opcode) */
166 else if (arg1 && arg2 && (PyLong_Check(arg1) || PyInt_Check(arg1)) && PyBytes_Check(arg2)) {
167 triton::uint64 addr = PyLong_AsUint64(arg1);
168 triton::uint8* opc = reinterpret_cast<triton::uint8*>(PyBytes_AsString(arg2));
169 triton::uint32 size = static_cast<triton::uint32>(PyBytes_Size(arg2));
170 return PyInstruction(addr, opc, size);
171 }
172
173 else {
174 return PyErr_Format(PyExc_TypeError, "Instruction(): Expects bytes as first argument or an integer as first and bytes as second argument.");
175 }
176 }
177 catch (const triton::exceptions::Exception& e) {
178 return PyErr_Format(PyExc_TypeError, "%s", e.what());
179 }
180 }
