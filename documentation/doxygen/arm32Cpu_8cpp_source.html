245 void Arm32Cpu::disassembly(triton::arch::Instruction& inst) {
246 triton::extlibs::capstone::csh handle;
247 triton::extlibs::capstone::cs_insn* insn;
248 triton::usize count = 0;
249
250 /* Check if the opcode and opcode' size are defined */
251 if (inst.getOpcode() == nullptr || inst.getSize() == 0)
252 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Opcode and opcodeSize must be definied.");
253
254 /* Select capstone handler (based on execution mode) */
255 handle = (this->thumb ? this->handleThumb : this->handleArm);
256
257 /* Clear instructicon's operands if alredy defined */
258 inst.operands.clear();
259
260 /* Update instruction address if undefined */
261 if (!inst.getAddress()) {
262 inst.setAddress(static_cast<triton::uint64>(this->getConcreteRegisterValue(this->getProgramCounter())));
263 }
264
265 /* Let's disass and build our operands */
266 count = triton::extlibs::capstone::cs_disasm(handle, inst.getOpcode(), inst.getSize(), inst.getAddress(), 0, &insn);
267 if (count > 0) {
268 triton::extlibs::capstone::cs_detail* detail = insn->detail;
269
270 /* Refine the opcode */
271 inst.setOpcode(insn[0].bytes, insn[0].size);
272
273 /* Refine the size */
274 inst.setSize(insn[0].size);
275
276 /* Init the instruction's type */
277 inst.setType(this->capstoneInstructionToTritonInstruction(insn[0].id));
278
279 /* Init the instruction's code codition */
280 inst.setCodeCondition(this->capstoneConditionToTritonCondition(detail->arm.cc));
281
282 /* Init the instruction's write back flag */
283 inst.setWriteBack(detail->arm.writeback);
284
285 /* Set True if the instruction udpate flags */
286 inst.setUpdateFlag(detail->arm.update_flags);
287
288 /* Set thumb mode */
289 inst.setThumb(thumb);
290
291 /* Set architecture */
292 inst.setArchitecture(triton::arch::ARCH_ARM32);
293
294 /* Init the disassembly */
295 std::stringstream str;
296
297 /* Add mnemonic */
298 str << insn[0].mnemonic;
299
300 /* Add operands */
301 if (inst.getType() == ID_INS_IT || detail->arm.op_count)
302 str << " " << insn[0].op_str;
303
304 inst.setDisassembly(str.str());
305
306 /* Process IT instruction */
307 if (inst.getType() == ID_INS_IT) {
308 /* Nested IT instruction, throw an exception as this is not valid ARM code */
309 if (this->itInstrsCount > 0)
310 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Nested IT instructions are not allowed.");
311
312 /* Copy state from the mnemonic of the instruction */
313 strncpy(this->itStateArray, &insn[0].mnemonic[1], 5);
314 this->itStateArray[4] = 0;
315
316 this->itInstrsCount = strlen(this->itStateArray);
317 this->itInstrIndex = 0;
318
319 this->itCC = inst.getCodeCondition();
320 this->itCCInv = this->invertCodeCondition(this->itCC);
321 }
322
323 /* Process instruction within an IT block */
324 if (inst.getType() != ID_INS_IT && this->itInstrsCount > 0) {
325 /* NOTE Assuming that CS always returns mnemonics in lower case */
326 triton::arch::arm::condition_e cc = this->itStateArray[this->itInstrIndex] == 't' ? this->itCC : this->itCCInv;
327
328 inst.setCodeCondition(cc);
329
330 this->itInstrsCount--;
331 this->itInstrIndex++;
332 }
333
334 /* Init operands */
335 for (triton::uint32 n = 0; n < detail->arm.op_count; n++) {
336 triton::extlibs::capstone::cs_arm_op* op = &(detail->arm.operands[n]);
337 switch(op->type) {
338
339 case triton::extlibs::capstone::ARM_OP_IMM: {
340 triton::arch::Immediate imm(op->imm, triton::size::dword);
341
342 if (op->subtracted)
343 imm.setSubtracted(true);
344
345 inst.operands.push_back(triton::arch::OperandWrapper(imm));
346 break;
347 }
348
349 case triton::extlibs::capstone::ARM_OP_MEM: {
350 triton::arch::MemoryAccess mem;
351
352 /* Set the size of the memory access */
353 mem.setBits(triton::bitsize::dword-1, 0);
354
355 /* LEA if exists */
356 const triton::arch::Register base(*this, this->capstoneRegisterToTritonRegister(op->mem.base));
357 triton::arch::Register index(*this, this->capstoneRegisterToTritonRegister(op->mem.index));
358
359 /* Set Shift type and value */
360 triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
361
362 index.setShiftType(shiftType);
363
364 switch(shiftType) {
365 case triton::arch::arm::ID_SHIFT_INVALID:
366 break;
367 case triton::arch::arm::ID_SHIFT_ASR:
368 case triton::arch::arm::ID_SHIFT_LSL:
369 case triton::arch::arm::ID_SHIFT_LSR:
370 case triton::arch::arm::ID_SHIFT_ROR:
371 index.setShiftValue(op->shift.value);
372 break;
373 case triton::arch::arm::ID_SHIFT_RRX:
374 /* NOTE: According to the manual RRX there is no
375 * immediate associated with this shift type. However,
376 * from the description of the instruction it can be
377 * deduced that a value of one is used.
378 */
379 index.setShiftValue(1);
380 break;
381 case triton::arch::arm::ID_SHIFT_ASR_REG:
382 case triton::arch::arm::ID_SHIFT_LSL_REG:
383 case triton::arch::arm::ID_SHIFT_LSR_REG:
384 case triton::arch::arm::ID_SHIFT_ROR_REG:
385 index.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
386 break;
387 case triton::arch::arm::ID_SHIFT_RRX_REG:
388 /* NOTE: Capstone considers this as a viable shift operand
389 * but according to the ARM manual this is not possible.
390 */
391 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
392 break;
393 default:
394 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
395 }
396
397 if (op->subtracted) {
398 index.setSubtracted(true);
399 }
400
401 triton::uint32 immsize = (
402 this->isRegisterValid(base.getId()) ? base.getSize() :
403 this->isRegisterValid(index.getId()) ? index.getSize() :
404 this->gprSize()
405 );
406
407 triton::arch::Immediate disp(op->mem.disp, immsize);
408
409 /* Specify that LEA contains a PC relative */
410 if (base.getId() == this->pcId) {
411 /* NOTE: PC always points to the address to the current
412 * instruction plus: a) 8 in case of ARM mode, or b) 4 in
413 * case of Thumb. It is also aligned to 4 bytes. For more
414 * information, refer to section "Use of labels in UAL
415 * instruction syntax" of the reference manual.
416 */
417 auto offset = this->thumb ? 4 : 8;
418 auto address = (inst.getAddress() + offset) & 0xfffffffc;
419 mem.setPcRelative(address);
420 }
421
422 /* Note that in ARM32 there is no segment register and scale value */
423 mem.setBaseRegister(base);
424 mem.setIndexRegister(index);
425 mem.setDisplacement(disp);
426
427 /* If there is an index register available, set scale to 1 to perform this following computation (base) + (index * scale) */
428 if (this->isRegisterValid(index.getId()))
429 mem.setScale(triton::arch::Immediate(1, index.getSize()));
430
431 inst.operands.push_back(triton::arch::OperandWrapper(mem));
432 break;
433 }
434
435 case triton::extlibs::capstone::ARM_OP_REG: {
436 triton::arch::Register reg(*this, this->capstoneRegisterToTritonRegister(op->reg));
437
438 /* Set Shift type and value */
439 triton::arch::arm::shift_e shiftType = this->capstoneShiftToTritonShift(op->shift.type);
440
441 reg.setShiftType(shiftType);
442
443 switch(shiftType) {
444 case triton::arch::arm::ID_SHIFT_INVALID:
445 break;
446 case triton::arch::arm::ID_SHIFT_ASR:
447 case triton::arch::arm::ID_SHIFT_LSL:
448 case triton::arch::arm::ID_SHIFT_LSR:
449 case triton::arch::arm::ID_SHIFT_ROR:
450 reg.setShiftValue(op->shift.value);
451 break;
452 case triton::arch::arm::ID_SHIFT_RRX:
453 /* NOTE: According to the manual RRX there is no
454 * immediate associated with this shift type. However,
455 * from the description of the instruction it can be
456 * deduced that a value of one is used.
457 */
458 reg.setShiftValue(1);
459 break;
460 case triton::arch::arm::ID_SHIFT_ASR_REG:
461 case triton::arch::arm::ID_SHIFT_LSL_REG:
462 case triton::arch::arm::ID_SHIFT_LSR_REG:
463 case triton::arch::arm::ID_SHIFT_ROR_REG:
464 reg.setShiftValue(this->capstoneRegisterToTritonRegister(op->shift.value));
465 break;
466 case triton::arch::arm::ID_SHIFT_RRX_REG:
467 /* NOTE: Capstone considers this as a viable shift operand
468 * but according to the ARM manual this is not possible.
469 */
470 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
471 break;
472 default:
473 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid shift type.");
474 }
475
476 if (op->subtracted)
477 reg.setSubtracted(true);
478
479 inst.operands.push_back(triton::arch::OperandWrapper(reg));
480 break;
481 }
482
483 default:
484 /* NOTE: FP, CIMM, and missing one are not supported yet. */
485 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Invalid operand.");
486 } // switch
487 } // for operand
488
489 /* Set branch */
490 if (detail->groups_count > 0) {
491 for (triton::uint32 n = 0; n < detail->groups_count; n++) {
492 if (detail->groups[n] == triton::extlibs::capstone::ARM_GRP_JUMP) {
493 inst.setBranch(true);
494 inst.setControlFlow(true);
495 }
496 }
497 }
498
499 /* Post process instruction */
500 this->postDisassembly(inst);
501
502 /* Free capstone stuffs */
503 triton::extlibs::capstone::cs_free(insn, count);
504 }
505 else
506 throw triton::exceptions::Disassembly("Arm32Cpu::disassembly(): Failed to disassemble the given code.");
507
508 return;
509 }

512 void Arm32Cpu::postDisassembly(triton::arch::Instruction& inst) const {
513 /* Fix update flag */
514 /* NOTE: Quick (and super ugly) hack. Capstone is reporting
515 * update_flags equals true for ADC, RSC and SBC instruction when
516 * it shouldn't (it should only report true when the S suffix is
517 * present). This will be removed once Capstone is fixed. For more
518 * information, see: https://github.com/aquynh/capstone/issues/1568.
519 */
520 switch (inst.getType()) {
521 case ID_INS_ADC:
522 case ID_INS_RSC:
523 case ID_INS_SBC:
524 if (inst.getDisassembly().at(3) != 's')
525 inst.setUpdateFlag(false);
526 break;
527 }
528
529 /* Make implicit destination operand explicit */
530 /* NOTE: For some instructions the destination operand is
531 * optional (in which case the first source operand is used as
532 * destination). Capstone returns always all three operands for
533 * ARM instruction (i.e. make the destination operand explicit).
534 * However, it does not do the same for Thumb instructions. Here
535 * we make the destination operand explicit (in order to simplify
536 * the implementation of the semantics).
537 */
538 if (inst.isThumb() && inst.operands.size() == 2) {
539 triton::arch::OperandWrapper op(inst.operands[0]);
540
541 switch (inst.getType()) {
542 case ID_INS_ADC:
543 case ID_INS_ADD:
544 case ID_INS_AND:
545 case ID_INS_ASR:
546 case ID_INS_BIC:
547 case ID_INS_EOR:
548 case ID_INS_LSL:
549 case ID_INS_LSR:
550 case ID_INS_ORR:
551 case ID_INS_ROR:
552 case ID_INS_SBC:
553 case ID_INS_SUB:
554 inst.operands.insert(inst.operands.begin(), op);
555 break;
556 }
557 }
558
559 /* NOTE: If the instruction is POP and contains a PC register,
560 * we have to define the instruction as modifiying the control
561 * flow. See #945.
562 */
563 if (inst.getType() == ID_INS_POP) {
564 /* FIXME: Maybe the loop is useless if PC is always the last operand? */
565 for (auto& op : inst.operands) {
566 if (op.getType() == triton::arch::OP_REG && op.getConstRegister().getId() == this->pcId) {
567 inst.setControlFlow(true);
568 break;
569 }
570 }
571 }
572 }

618 triton::uint512 Arm32Cpu::getConcreteRegisterValue(const triton::arch::Register& reg, bool execCallbacks) const {
619 triton::uint512 value = 0;
620
621 if (execCallbacks && this->callbacks)
622 this->callbacks->processCallbacks(triton::callbacks::GET_CONCRETE_REGISTER_VALUE, reg);
623
624 switch (reg.getId()) {
625 case triton::arch::ID_REG_ARM32_R0: return (*((triton::uint32*)(this->r0)));
626 case triton::arch::ID_REG_ARM32_R1: return (*((triton::uint32*)(this->r1)));
627 case triton::arch::ID_REG_ARM32_R2: return (*((triton::uint32*)(this->r2)));
628 case triton::arch::ID_REG_ARM32_R3: return (*((triton::uint32*)(this->r3)));

698 void Arm32Cpu::setConcreteRegisterValue(const triton::arch::Register& reg, const triton::uint512& value) {
699 if (value > reg.getMaxValue())
700 throw triton::exceptions::Register("Arm32Cpu::setConcreteRegisterValue(): You cannot set this concrete value (too big) to this register.");
701
702 if (this->callbacks)
703 this->callbacks->processCallbacks(triton::callbacks::SET_CONCRETE_REGISTER_VALUE, reg, value);
704
705 switch (reg.getId()) {
706 case triton::arch::ID_REG_ARM32_R0: (*((triton::uint32*)(this->r0))) = static_cast<triton::uint32>(value); break;
707 case triton::arch::ID_REG_ARM32_R1: (*((triton::uint32*)(this->r1))) = static_cast<triton::uint32>(value); break;
708 case triton::arch::ID_REG_ARM32_R2: (*((triton::uint32*)(this->r2))) = static_cast<triton::uint32>(value); break;
709 case triton::arch::ID_REG_ARM32_R3: (*((triton::uint32*)(this->r3))) = static_cast<triton::uint32>(value); break;
710 case triton::arch::ID_REG_ARM32_R4: (*((triton::uint32*)(this->r4))) = static_cast<triton::uint32>(value); break;
711 case triton::arch::ID_REG_ARM32_R5: (*((triton::uint32*)(this->r5))) = static_cast<triton::uint32>(value); break;
712 case triton::arch::ID_REG_ARM32_R6: (*((triton::uint32*)(this->r6))) = static_cast<triton::uint32>(value); break;
713 case triton::arch::ID_REG_ARM32_R7: (*((triton::uint32*)(this->r7))) = static_cast<triton::uint32>(value); break;
714 case triton::arch::ID_REG_ARM32_R8: (*((triton::uint32*)(this->r8))) = static_cast<triton::uint32>(value); break;
715 case triton::arch::ID_REG_ARM32_R9: (*((triton::uint32*)(this->r9))) = static_cast<triton::uint32>(value); break;
716 case triton::arch::ID_REG_ARM32_R10: (*((triton::uint32*)(this->r10))) = static_cast<triton::uint32>(value); break;
717 case triton::arch::ID_REG_ARM32_R11: (*((triton::uint32*)(this->r11))) = static_cast<triton::uint32>(value); break;
718 case triton::arch::ID_REG_ARM32_R12: (*((triton::uint32*)(this->r12))) = static_cast<triton::uint32>(value); break;
719 case triton::arch::ID_REG_ARM32_SP: (*((triton::uint32*)(this->sp))) = static_cast<triton::uint32>(value); break;
720 case triton::arch::ID_REG_ARM32_R14: (*((triton::uint32*)(this->r14))) = static_cast<triton::uint32>(value); break;
721 case triton::arch::ID_REG_ARM32_PC: {
722 /* NOTE: Once in Thumb mode only switch to ARM through a Branch
723 * and Exchange instruction. The reason for this is that after
724 * switching to Thumb the ISB (instruction set selection bit) is
725 * cleared. Therefore, if we allow to switch back to ARM through
726 * these mechanism we would have a problem processing Thumb
727 * instructions.
728 */
729 auto pc = static_cast<triton::uint32>(value);
730 if (this->isThumb() == false && (pc & 0x1) == 0x1) {
731 this->setThumb(true);
732 }
733 (*((triton::uint32*)(this->pc))) = pc & ~0x1;
734 break;
735 }
736 case triton::arch::ID_REG_ARM32_APSR: (*((triton::uint32*)(this->apsr))) = static_cast<triton::uint32>(value); break;
737 case triton::arch::ID_REG_ARM32_N: {
738 triton::uint32 b = (*((triton::uint32*)(this->apsr)));
739 (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 31) : b & ~(1 << 31);
740 break;
741 }
742 case triton::arch::ID_REG_ARM32_Z: {
743 triton::uint32 b = (*((triton::uint32*)(this->apsr)));
744 (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 30) : b & ~(1 << 30);
745 break;
746 }
747 case triton::arch::ID_REG_ARM32_C: {
748 triton::uint32 b = (*((triton::uint32*)(this->apsr)));
749 (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 29) : b & ~(1 << 29);
750 break;
751 }
752 case triton::arch::ID_REG_ARM32_V: {
753 triton::uint32 b = (*((triton::uint32*)(this->apsr)));
754 (*((triton::uint32*)(this->apsr))) = !value.is_zero() ? b | (1 << 28) : b & ~(1 << 28);
755 break;
756 }
757 default:
758 throw triton::exceptions::Cpu("Arm32Cpu:setConcreteRegisterValue(): Invalid register.");
759 }
