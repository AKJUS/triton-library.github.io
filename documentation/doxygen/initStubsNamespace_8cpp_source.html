45 static PyObject* initCode(const std::vector<triton::uint8>& code) {
46 auto* area = new triton::uint8[code.size()];
47
48 for (triton::usize index = 0; index < code.size(); index++)
49 area[index] = code[index];
50
51 auto* ret = PyBytes_FromStringAndSize(reinterpret_cast<const char*>(area), code.size());
52 delete[] area;
53
54 return ret;
55 }

57 static PyObject* initSymbols(const std::map<std::string, triton::uint64>& symbols) {
58 PyObject* symbolsDict = xPyDict_New();
59 for (const auto& it : symbols) {
60 xPyDict_SetItem(symbolsDict, xPyString_FromString(it.first.c_str()), PyLong_FromUsize(it.second));
